DROP TABLE IF EXISTS doc;
CREATE TABLE `doc` (
`id` INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
`docFileNumber` INT( 5 ) NOT NULL,
`docDate` DATE NOT NULL,
`docApprover` VARCHAR( 40 ) NOT NULL,
`docArea` VARCHAR( 30 ) NOT NULL,
`docBlock` VARCHAR( 6 ) NOT NULL,
`docStreet` VARCHAR( 30 ) NOT NULL,
`docBuilding` VARCHAR( 6 ) NOT NULL,
`docPACINumber` INT( 8 ),
`docTitle` VARCHAR( 255 ) NOT NULL,
`docComment` VARCHAR( 20000 ),
`sectionId` TINYINT(1) NOT NULL DEFAULT 0,
`employeeId` VARCHAR( 20 ) DEFAULT NULL,
`time` TIMESTAMP
);
ALTER TABLE `mewdesigncomps`.`doc` ADD UNIQUE INDEX `file_number_UNIQUE` (`docFileNumber` ASC) ;
